#include <iostream>
#include "mat.h"
using std::cin; using std::cout; using std::endl; 
struct Matrix {
    int a, b, c, d; // Representing the 2x2 matrix as 4 variables
};
          // A = A + B, and return reference to A
Matrix& addTo(Matrix& A, const Matrix& B) {
    A.a += B.a;
    A.b += B.b;
    A.c += B.c;
    A.d += B.d;
    return A;
}

// Function to read a matrix
Matrix readMatrix() {
    Matrix mat;
    cin >> mat.a >> mat.b >> mat.c >> mat.d;
    return mat;
}

// Function to print a matrix
void printMatrix(const Matrix &mat) {
    cout << mat.a << " " << mat.b << endl;
    cout << mat.c << " " << mat.d << endl;
}
int main() {
    Matrix A = readMatrix();
    Matrix B = readMatrix();
    Matrix C = readMatrix();
    addTo(addTo(A, B),C);  // A = A + B + C
    printMatrix(A);  // Final result is stored in A
}
